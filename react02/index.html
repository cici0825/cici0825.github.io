import {useState,useRef} from 'react'
import {BlackTodoItem} from './TodoItem'

//針對 todoItem restyle 傳進todoitem後要在原本的 function 給 className 的值 並把 className 屬性掛在指定改變樣式的block
// let id = 2
function App() {
  
  const [todos,setTodos]= useState([
     {id:1,content:'abc'}
  ])   
  const [value,setValue] = useState('') 
  const id = useRef(2)
  const  handelButtonClick = () => {
    setTodos([{
      id:id.current,
      content:value
    },
    ...todos])//把state的值複製過來(淺拷貝)，傳進新的參數
    setValue('')
    id.current++
  }
  const handleInputChange = (e) => {
    setValue(e.target.value)
  }
  const handleDeleteTodo = id => {
    setTodos(todos.filter(todo =>todo.id !== id)) 
    //true 會留下 => ex [1,2,3].filter(i=>i>1) 2,3會留下
    // 如果todo的id !== 我按到的 id 就留下,所以被按到的會刪除 
  }
  return(
    <div className="APP">
      <input  type = "text" placeholder = "add todo..." value = {value} onChange = {handleInputChange} />
      <button onClick = {handelButtonClick} > Add Todo </button>
      {
        todos.map(todo => < BlackTodoItem key = {todo.id} todo={todo} handleDeleteTodo={handleDeleteTodo} />)
      }   
    </div>
  );
}
export default App;
